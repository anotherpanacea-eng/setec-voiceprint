# Acquisition-only dependencies for SETEC's impostor-corpus tooling.
#
# Install on demand:
#   pip install -r requirements-acquisition.txt
#
# These are required by `scripts/acquire_blog.py`,
# `scripts/acquire_blogger_takeout.py`, `scripts/acquire_magazine.py`,
# `scripts/pdf_inventory.py`, and `scripts/pdf_extract.py` only.
# Ordinary SETEC variance audits, voice distance, validation harness,
# and stylometry oracle runs do NOT need them. The core requirements
# live in `requirements.txt`; this layer is opt-in like
# `requirements-calibration.txt`.
#
# Acquisition tools acquire long-form prose from third-party sources
# (Substacks, WordPress / Ghost blogs, generic HTML archives, online
# magazines, and existing PDF libraries) for the impostor corpus that
# the future General Imposters validation harness consumes. See
# `internal/2026-05-08-impostor-corpus-spec.md` and
# `references/manifest-schema.md` for context. Per-script provenance
# is recorded in each acquired entry's `acquired_via` manifest field.
#
# Network tools honor robots.txt by default; v1 ships no override
# flag. Acquired text is stored under the same private directory as
# the user's own baselines (`ai-prose-baselines-private/impostors/`)
# and is never redistributed.

# HTTP client. Used by the network-bound acquisition scripts for feed,
# sitemap, archive, and post fetches.
requests>=2.31

# RSS / Atom feed parsing. Substack and WordPress / Ghost expose feeds
# at `<url>/feed`, `/feed/`, or `/feed/atom/`; feedparser handles all
# three uniformly and exposes per-entry `content`, `summary`, `link`,
# and `published` fields.
feedparser>=6.0

# HTML parsing for blog and magazine post bodies. lxml is the parser
# backend BeautifulSoup uses for production HTML; faster and more
# tolerant than the stdlib `html.parser` fallback.
beautifulsoup4>=4.12
lxml>=4.9

# Primary main-content extractor (Apache-2.0). trafilatura's model-free
# readability heuristics strip boilerplate/nav/comments and isolate the
# article body, replacing per-site `content_selector` / `strip_selectors`
# tuning for the common case (`acquisition_core.extract_main_content`).
# OPTIONAL within this tier: extraction fail-softs to the BeautifulSoup
# path (`html_to_text`) when trafilatura is absent or returns nothing, so
# acquisition still runs without it — it is an upgrade, not a hard dep.
trafilatura>=1.8

# Date parsing across feed and HTML formats. RSS / Atom feeds emit
# RFC822 dates; magazine archives emit a mix of ISO and human-readable
# dates. python-dateutil normalizes all of those to `datetime.date`.
python-dateutil>=2.8

# PDF inventory and text-layer extraction (commit 3 of the impostor-
# corpus spec). Lighter than `pdfminer.six` for the inventory pass and
# robust enough for text extraction on academic papers.
pypdf>=4.0

# MinHash-LSH near-duplicate dedup (MIT) for the staged acquisition
# manifest (`scripts/near_dup_dedup.py`). Removes near-identical reposts /
# reprints / cross-source pulls that the exact SHA-256 dedup can't see,
# before the final JSONL manifest is committed. OPTIONAL within this tier:
# the dedup pass is opt-in and lazy-imports datasketch, so base imports
# stay clean and ordinary acquisition runs without it. Pulls numpy/scipy.
datasketch>=1.6

# Optional: Wayback Machine access for shut-down blogs (Slate Star
# Codex archive, defunct Substacks). The `wayback` package wraps the
# CDX API. Acquisition scripts handle ImportError gracefully — if you
# don't pass `--wayback`, you don't need this dependency.
# wayback>=0.4

# OCR support for image-only PDFs (commit 3 of the impostor-corpus
# spec). `ocrmypdf` requires system binaries (tesseract, ghostscript,
# qpdf) installed separately:
#
#   brew install tesseract ghostscript qpdf
#
# Acquisition scripts emit a clear "install OCR dependencies or pass
# --skip-ocr" message when ocrmypdf is unavailable.
# ocrmypdf>=15

# AWS S3 client for the Mirrulations regulatory-comment acquirer
# (`scripts/acquire_mirrulations.py`). Used for anonymous (unsigned) reads
# of the public Mirrulations Open Data bucket; not needed by any other
# acquisition script. The store abstraction falls back to a clear
# "boto3 is not installed" message when absent.
boto3>=1.28
